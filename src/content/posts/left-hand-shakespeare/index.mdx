import { Scrawl, ScrawlGrid } from "~/components/Scrawl";
import quickLeft from "./quick-left.scrawl?raw";
import quickRight from "./quick-right.scrawl?raw";

export const texts = Object.values(
  import.meta.glob("./sonnets/text/*.txt", {
    query: "?raw",
    import: "default",
    eager: true,
  }),
);
export const sonnets = Object.values(
  import.meta.glob("./sonnets/scrawl/*.scrawl", {
    query: "?url",
    import: "default",
    eager: true,
  }),
).map((scrawlUrl, i) => ({
  scrawlUrl,
  text: texts[i],
}));

The ability to write with either hand seems useful, so I'm writing Shakespeare's sonnets for practice. Along the way, I'm building the tool to animate my writing, available at [https://rescrawl.kylezhe.ng](https://rescrawl.kylezhe.ng).

Here is my handwriting from March, 17, 2026, the start of this project.

<span class="text-sm font-bold">Right hand</span>
<Scrawl data={quickRight} width={800} height={100} client:visible />
<span class="text-sm font-bold">Left hand</span>
<Scrawl data={quickLeft} width={800} height={100} client:visible />

It's slow, messy, and sometimes my hand spasms from exertion. I'm using muscles I've never moved, writing words I have never read. Bear witness to my improvement below.
